  public static void addSpell(Spell spell) {
    spells.put(Integer.valueOf(spell.getNumber()), spell);
    if (spell.isTargetCreature())
      creatureSpells.put(Integer.valueOf(spell.getNumber()), spell); 
    if (spell.isTargetAnyItem())
      itemSpells.put(Integer.valueOf(spell.getNumber()), spell); 
    if (spell.isTargetWound())
      woundSpells.put(Integer.valueOf(spell.getNumber()), spell); 
    if (spell.isTargetTile())
      tileSpells.put(Integer.valueOf(spell.getNumber()), spell); 
    if (spell.getEnchantment() != 0)
      enchantments.put(Byte.valueOf(spell.getEnchantment()), spell); 
  }
  
  public static final Spell getSpell(int number) {
    return spells.get(Integer.valueOf(number));
  }
  
  public static final Spell[] getAllSpells() {
    return (Spell[])spells.values().toArray((Object[])new Spell[spells.size()]);
  }
  
  public static final Spell getEnchantment(byte num) {
    return enchantments.get(Byte.valueOf(num));
  }
  
  public static final Spell[] getSpellsTargettingItems() {
    Set<Spell> toReturn = new HashSet<>();
    for (Iterator<Integer> it = itemSpells.keySet().iterator(); it.hasNext();)
      toReturn.add(itemSpells.get(it.next())); 
    return toReturn.<Spell>toArray(new Spell[toReturn.size()]);
  }
  
  public static final Spell[] getSpellsEnchantingItems() {
    Set<Spell> toReturn = new HashSet<>();
    for (Iterator<Integer> it = itemSpells.keySet().iterator(); it.hasNext(); ) {
      Spell spell = itemSpells.get(it.next());
      if (spell.getEnchantment() > 0)
        toReturn.add(spell); 
    } 
    return toReturn.<Spell>toArray(new Spell[toReturn.size()]);
  }
